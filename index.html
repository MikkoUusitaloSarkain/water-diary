Admin settings

Database settings are hidden from normal use. Use this only if you need to change the Supabase project URL or publishable key.

Use this table/policy model for the secure authenticated version:

create table if not exists water_entries ( id uuid primary key default gen_random_uuid(), user_id uuid not null references auth.users(id) on delete cascade, entry_date date not null, entry_time time not null, amount_ml integer not null check (amount_ml > 0), drink_type text not null default 'water', note text, created_at timestamptz not null default now() ); alter table water_entries enable row level security; drop policy if exists "Allow anon read" on water_entries; drop policy if exists "Allow anon insert" on water_entries; drop policy if exists "Allow anon delete" on water_entries; drop policy if exists "Users can read own entries" on water_entries; drop policy if exists "Users can insert own entries" on water_entries; drop policy if exists "Users can delete own entries" on water_entries; create policy "Users can read own entries" on water_entries for select using (auth.uid() = user_id); create policy "Users can insert own entries" on water_entries for insert with check (auth.uid() = user_id); create policy "Users can delete own entries" on water_entries for delete using (auth.uid() = user_id);
